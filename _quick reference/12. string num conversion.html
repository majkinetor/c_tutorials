C String Conversion Quick Reference

Quick reference for atoi, atof, and sscanf functions.

atoi / atol - String to Integer

Header

#include <stdlib.h>

Function Signatures

int atoi(const char *str);
long atol(const char *str);

Basic Examples

#include <stdio.h>
#include <stdlib.h>

int main() {
    char *str1 = "42";
    char *str2 = "-123";
    char *str3 = "  456";    // Leading whitespace OK
    char *str4 = "789abc";   // Stops at first non-digit
    
    printf("atoi(\"%s\") = %d\n", str1, atoi(str1));  // 42
    printf("atoi(\"%s\") = %d\n", str2, atoi(str2));  // -123
    printf("atoi(\"%s\") = %d\n", str3, atoi(str3));  // 456
    printf("atoi(\"%s\") = %d\n", str4, atoi(str4));  // 789
    
    return 0;
}

atof - String to Floating Point

Function Signature

double atof(const char *str);

Examples

#include <stdio.h>
#include <stdlib.h>

int main() {
    char *str1 = "3.14159";
    char *str2 = "-2.5";
    char *str3 = "1.23e-4";  // Scientific notation
    char *str4 = "  42.7";   // Leading whitespace
    
    printf("atof(\"%s\") = %f\n", str1, atof(str1));  // 3.141590
    printf("atof(\"%s\") = %f\n", str2, atof(str2));  // -2.500000
    printf("atof(\"%s\") = %e\n", str3, atof(str3));  // 1.230000e-04
    printf("atof(\"%s\") = %.1f\n", str4, atof(str4)); // 42.7
    
    return 0;
}

sscanf - Formatted String Parsing

Header

#include <stdio.h>

Function Signature

int sscanf(const char *str, const char *format, ...);
Return Value: Number of successfully parsed items, or EOF on error.

Basic Parsing Examples

#include <stdio.h>

int main() {
    int num;
    float fval;
    char word[50];
    
    // Parse single integer
    sscanf("42", "%d", &num);
    printf("Integer: %d\n", num);  // 42
    
    // Parse float
    sscanf("3.14", "%f", &fval);
    printf("Float: %.2f\n", fval);  // 3.14
    
    // Parse string (stops at whitespace)
    sscanf("Hello", "%s", word);
    printf("Word: %s\n", word);  // Hello
    
    return 0;
}

Multiple Values

int main() {
    int a, b, c;
    char name[50];
    float price;
    
    // Parse multiple integers
    sscanf("10 20 30", "%d %d %d", &a, &b, &c);
    printf("%d, %d, %d\n", a, b, c);  // 10, 20, 30
    
    // Mixed types
    sscanf("Apple 2.99", "%s %f", name, &price);
    printf("%s costs $%.2f\n", name, price);  // Apple costs $2.99
    
    return 0;
}

Format Specifiers

Specifier Type Description
%d int* Decimal integer
%i int* Integer (auto-detect base)
%u unsigned int* Unsigned decimal
%f, %e, %g float* Floating point
%lf double* Double precision
%s char* String (stops at whitespace)
%c char* Single character
%x unsigned int* Hexadecimal
%o unsigned int* Octal

Advanced Patterns

int main() {
    char str[100];
    int day, month, year;
    int hex_val;
    char c;
    
    // Parse date format
    sscanf("2024-12-25", "%d-%d-%d", &year, &month, &day);
    printf("Date: %02d/%02d/%d\n", day, month, year);
    
    // Parse hexadecimal
    sscanf("0xFF", "%x", &hex_val);
    printf("Hex value: %d\n", hex_val);  // 255
    
    // Width specifier - read only N characters
    sscanf("HelloWorld", "%5s", str);
    printf("First 5 chars: %s\n", str);  // Hello
    
    // Skip characters with *
    sscanf("abc:123", "%*[^:]:%d", &day);  // Skip until ':', then parse int
    printf("Number: %d\n", day);  // 123
    
    return 0;
}

Checking Return Value

int main() {
    int x, y;
    int result;
    
    // Good practice: check how many items were parsed
    result = sscanf("42 56", "%d %d", &x, &y);
    if (result == 2) {
        printf("Parsed both: %d and %d\n", x, y);
    } else {
        printf("Parse failed, got %d items\n", result);
    }
    
    // Check for partial parse
    result = sscanf("123 abc", "%d %d", &x, &y);
    if (result != 2) {
        printf("Only parsed %d items\n", result);  // Only parsed 1 items
    }
    
    return 0;
}

Character Class Matching

int main() {
    char buffer[100];
    
    // %[...] - match character set
    sscanf("abc123", "%[abc]", buffer);
    printf("Matched: %s\n", buffer);  // abc
    
    // %[^...] - match anything EXCEPT character set
    sscanf("hello world", "%[^ ]", buffer);
    printf("Until space: %s\n", buffer);  // hello
    
    // Parse until comma
    sscanf("John,Doe", "%[^,]", buffer);
    printf("First name: %s\n", buffer);  // John
    
    return 0;
}
