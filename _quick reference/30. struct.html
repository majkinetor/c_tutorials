C Struct Quick Reference

Basic Declaration and Definition

// Declaration
struct Point {
    int x;
    int y;
};

// Declaration with typedef
typedef struct {
    int x;
    int y;
} Point;

// Named struct with typedef
typedef struct Point {
    int x;
    int y;
} Point;

Initialization

// Designated initializers (C99+)
struct Point p1 = {.x = 10, .y = 20};
struct Point p2 = {.y = 30};  // x is 0

// Positional initialization
struct Point p3 = {10, 20};

// Zero initialization
struct Point p4 = {0};

// Array of structs
struct Point points[] = {
    {1, 2},
    {3, 4},
    {.x = 5, .y = 6}
};

Member Access

// Direct access
struct Point p = {10, 20};
int x_val = p.x;
p.y = 30;

// Pointer access
struct Point *ptr = &p;
int x_val = ptr->x;        // Arrow operator
int y_val = (*ptr).y;      // Equivalent dereference notation

Nested Structs

struct Rectangle {
    struct Point top_left;
    struct Point bottom_right;
};

struct Rectangle r = {{0, 10}, {20, 0}};
int x = r.top_left.x;

struct Rectangle *rptr = &r;
int y = rptr->bottom_right.y;

Copying

// Struct assignment (shallow copy)
struct Point p1 = {10, 20};
struct Point p2 = p1;  // All members copied

// Manual copy
memcpy(&p2, &p1, sizeof(struct Point));

Function Parameters

// Pass by value (copy)
void modify_copy(struct Point p) {
    p.x = 100;  // Original unchanged
}

// Pass by pointer (can modify)
void modify_original(struct Point *p) {
    p->x = 100;  // Original changed
}

// Const pointer (read-only)
void read_only(const struct Point *p) {
    int x = p->x;  // OK
    // p->x = 100;  // Compilation error
}

Dynamic Allocation

// Single struct
struct Point *p = malloc(sizeof(struct Point));
p->x = 10;
p->y = 20;
free(p);

// Array of structs
struct Point *arr = malloc(10 * sizeof(struct Point));
arr[0].x = 1;
free(arr);

// Always check malloc return
if (p == NULL) {
    // Handle allocation failure
}

Key Notes

Important Points:
  • Structs cannot contain instances of themselves, only pointers
  • Padding between members depends on alignment requirements
  • Assignment copies all members (shallow copy for pointers)

  • Initialization sets unspecified members to zero
  • Arrow operator -> is shorthand for (*ptr).member
